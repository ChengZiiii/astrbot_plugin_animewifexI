# app/storage/stores.py — 追加
class PkBattleStore:
    def __init__(self, base_dir: Path):
        self.base = base_dir
    def _path(self, gid, battle_id):
        return self.base / "data" / "groups" / gid / "pk_battles" / f"{battle_id}.json"
    def save(self, gid, battle_id, battle: PkBattle):
        path = self._path(gid, battle_id)
        path.parent.mkdir(parents=True, exist_ok=True)
        atomic_write_json(path, battle.to_dict())
    def load(self, gid, battle_id) -> Optional[PkBattle]:
        path = self._path(gid, battle_id)
        if not path.exists(): return None
        return PkBattle.from_dict(load_json(path))
    def list_active(self, gid) -> List[PkBattle]:
        d = self.base / "data" / "groups" / gid / "pk_battles"
        if not d.exists(): return []
        return [PkBattle.from_dict(load_json(p)) for p in d.glob("*.json")]
    def delete(self, gid, battle_id):
        path = self._path(gid, battle_id)
        if path.exists(): path.unlink()
```

    rarity, elem, mods = combos[(0, 2)]  # SSR, 智力

# app/services/pk_v2_passives.py
RARITY_PASSIVES = {...}  # SSR/SR/R/N
ELEMENT_PASSIVES = {...} # 力量/敏捷/智力

def compute_combined_modifiers(rarity: str, element: str) -> dict:
    mods = {}
    if rarity in RARITY_PASSIVES:
        mods.update(RARITY_PASSIVES[rarity]["modifier"])
    if element in ELEMENT_PASSIVES:
        mods.update(ELEMENT_PASSIVES[element]["modifier"])
    return mods
```
